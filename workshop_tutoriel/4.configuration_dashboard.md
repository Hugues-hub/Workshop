```php
use App\User;
use Illuminate\Support\Facades\Auth;
...
public function index()
{
    $startUp = Auth::user()->start_up;
    $users = User::where('start_up', $startUp)->get();
    $name = Auth::user()->name;
    return view(
        'home',
        [
          'name' => $name,
          'users' => $users,
          'startUp' => $startUp
        ]
    );
}
```
